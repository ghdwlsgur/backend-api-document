firebase login --reauth // 파이어베이스 재인증
firebase use hello-buenos // pro 데이터베이스 선택
firebase use hello-dev-beb37 // dev 데이터베이스 선택
yarn deployd // dev 서버 배포
yarn deployp // pro 서버 배포
yarn testd index.test.js // index.test.js 단위테스트 (dev 서버)
yarn testp index.test.js // index.test.js 단위테스트 (pro 서버)
